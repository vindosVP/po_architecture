```plantuml
@startuml
!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

Person(admin, "Администратор")

System(conference_site, "Сайт сервиса доставок", "Веб-сайт для работы с доставками")


Rel(admin, conference_site, "Просмотр, добавление и редактирование информации о пользователях, посылках и доставках")
Rel(user, conference_site, "Создание, просмотр информации о посылках, доставках и их оформление")
@enduml
```
